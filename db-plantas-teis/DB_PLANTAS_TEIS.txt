-- 2.1. Tabla FamiliaBotanica
--
-- Relaciona las plantas con las familias botánicas, almacenando
-- información como el nombre, descripción, origen geográfico, 
-- número de especies, y estado de conservación.
--
--		idFamilia			Identificador único de la familia botánica
--		nombre 				Nombre de la familia botánica (ej. "Solanaceae")
--		nombreComun			Nombre común de la familia botánica si tiene uno (ej. "Tomates" para Solanaceae)
--		descripcion			Descripción detallada de la familia (características generales)
--		origen				Región o área geográfica de la familia (ej. "América del Sur")
--		numEspecies			Número de especies dentro de la familia
--		caracteristicas		Aspectos distintivos de la familia (ej. "Plantas herbáceas, con hojas alternas")
--		estadoConservacion	Tipo de conservación de la familia botánica (ej. "En peligro", "Preocupación menor")

CREATE TABLE FamiliaBotanica (
    idFamilia INT,
    nombreFamilia varchar (100),
    nombreComum varchar (100), 			-- no pueda ser nulo
    descripcion varchar (255),			-- que tenga contenido
    origen varchar (100),				-- una provincia gallega
    numEspecies int, 					-- valor natural
    caracteristicas varchar (255), 		-- que contenido
    estadoConservacion varchar (50),	-- que sea un valor
    
    CONSTRAINT PK_FamiliaBotanica					PRIMARY KEY (idFamilia),
    CONSTRAINT UQ_FamiliaBotanica_nombre_familia	UNIQUE (nombreFamilia),
    
    CONSTRAINT CHK_nombreComun						CHECK (nombreComun IS NOT NULL),
    CONSTRAINT CHK_descripcion_nula					CHECK (descripcion IS NOT NULL),
    CONSTRAINT CHK_descripcion_vacia				CHECK (descripcion != ''),
    CONSTRAINT CHK_origen_desconocido				CHECK (origen IN ('A Coruña', 'Lugo', 'Ourense', 'Pontevedra' )), -- que sea de galicia
    CONSTRAINT CHK_numEspecies_invalido				CHECK (numEspecies > 0 AND numEspecies < 5000)
 	
    );
